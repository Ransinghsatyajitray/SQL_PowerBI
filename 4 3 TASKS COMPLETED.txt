TASK 1: HOW TO REPORT LIST OF ALL EMPLOYEES WITH MAXIMUM SALARY?
        SELECT EmployeeName FROM EMPLOYEE_INFO WHERE EmployeeSalary= (SELECT MAX(EmployeeSalary) FROM EMPLOYEE_INFO)

TASK 2: HOW TO REPORT LIST OF ALL EMPLOYEES WITH MINIMUM SALARY?
        SELECT EmployeeName FROM EMPLOYEE_INFO WHERE EmployeeSalary= (SELECT MIN(EmployeeSalary) FROM EMPLOYEE_INFO)

TASK 3: HOW TO REPORT LIST OF ALL EMPLOYEES WITH MINIMUM OR MAXIMUM SALARY?
        SELECT EmployeeName FROM EMPLOYEE_INFO WHERE EmployeeSalary= (SELECT MIN(EmployeeSalary) FROM EMPLOYEE_INFO) OR EmployeeSalary= (SELECT MAX(EmployeeSalary) FROM EMPLOYEE_INFO)

TASK 4: HOW TO REPORT LIST OF ALL EMPLOYEES WITH MINIMUM OR MAXIMUM SALARY IN THE ORDER OF THEIR EMPLOYEE IDs?
        SELECT EmployeeName FROM EMPLOYEE_INFO WHERE EmployeeSalary= (SELECT MIN(EmployeeSalary) FROM EMPLOYEE_INFO) OR EmployeeSalary= (SELECT MAX(EmployeeSalary) FROM EMPLOYEE_INFO) ORDER BY EmployeeID ASC


TASK 5: HOW TO REPORT LIST OF ALL EMPLOYEES WITH MINIMUM SALARY FROM INDIA?
        SELECT EmployeeName FROM EMPLOYEE_INFO WHERE EmployeeSalary =(SELECT MIN(EmployeeSalary) FROM EMPLOYEE_INFO WHERE EmployeeCountry= 'INDIA')

TASK 7: HOW TO REPORT LIST OF ALL EMPLOYEES WITH MAXIMUM SALARY FROM INDIA WITH NAME STARTING WITH LETTER A ?
        SELECT EmployeeName FROM EMPLOYEE_INFO WHERE EmployeeSalary= (SELECT MAX(EmployeeSalary) FROM EMPLOYEE_INFO WHERE EmployeeCountry= 'INDIA' AND EmployeeName LIKE 'A%')
